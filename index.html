from js import document, setInterval from pyodide.ffi import create_proxy import random calm_words = ["spokój", "oddech", "tu i teraz", "równowaga"] bad_words = ["chaos", "stres", "pośpiech", "niepokój"] bg_color = [255, 243, 224] step_good = 15 step_bad = 5 def rgb_to_str(rgb): return f"rgb({rgb[0]},{rgb[1]},{rgb[2]})" def create_cloud(*args): global bg_color if random.random() < 0.75: word = random.choice(calm_words) is_calm = True else: word = random.choice(bad_words) is_calm = False cloud = document.createElement("div") cloud.classList.add("cloud") cloud.innerText = word cloud.style.left = f"{random.randint(0, 85)}%" document.body.appendChild(cloud) def remove(event=None): if cloud in document.body.children: document.body.removeChild(cloud) def on_click(event): global bg_color step = step_good if is_calm else step_bad if is_calm: target = [173, 216, 230] # błękit else: target = [255, 100, 100] # czerwony for i in range(3): if bg_color[i] < target[i]: bg_color[i] = min(bg_color[i]+step, target[i]) elif bg_color[i] > target[i]: bg_color[i] = max(bg_color[i]-step, target[i]) document.body.style.background = rgb_to_str(bg_color) remove() cloud.addEventListener("click", create_proxy(on_click)) cloud.addEventListener("animationend", create_proxy(remove)) create_cloud_proxy = create_proxy(create_cloud) setInterval(create_cloud_proxy, 800) # częstsze obłoki
