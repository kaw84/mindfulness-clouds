from js import document, setInterval from pyodide.ffi import create_proxy import random words = [ "spokój", "oddech", "tu i teraz", "równowaga", "świadomość", "wdzięczność", "akceptacja", "cisza" ] def create_cloud(*args): word = random.choice(words) cloud = document.createElement("div") cloud.classList.add("cloud") cloud.innerText = word cloud.style.left = f"{random.randint(0, 85)}%" document.body.appendChild(cloud) # usuń po zakończeniu animacji lub kliknięciu def remove(event=None): if cloud in document.body.children: document.body.removeChild(cloud) # po kliknięciu cloud.addEventListener("click", create_proxy(remove)) # po zakończeniu animacji cloud.addEventListener("animationend", create_proxy(remove)) # Proxy funkcji i interwał generowania obłoków create_cloud_proxy = create_proxy(create_cloud) setInterval(create_cloud_proxy, 2000)
